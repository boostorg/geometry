00119     static inline return_type segments_intersect(side_info const& sides,
00120                     coordinate_type const& dx1, coordinate_type const& dy1,
00121                     coordinate_type const& dx2, coordinate_type const& dy2,
00122                     S1 const& s1, S2 const& s2)
00123     {
00124         bool const ra0 = sides.get<0,0>() == 0;
00125         bool const ra1 = sides.get<0,1>() == 0;
00126         bool const rb0 = sides.get<1,0>() == 0;
00127         bool const rb1 = sides.get<1,1>() == 0;
00128 
00129         return
00130             // opposite and same starting point (FROM)
00131             ra0 && rb0 ? calculate_side<1>(sides, dx1, dy1, s1, s2, 'f', -1, -1)
00132 
00133             // opposite and point to each other (TO)
00134             : ra1 && rb1 ? calculate_side<0>(sides, dx1, dy1, s1, s2, 't', 1, 1)
00135 
00136             // not opposite, forming an angle, first a then b,
00137             // directed either both left, or both right
00138             // Check side of B2 from A. This is not calculated before
00139             : ra1 && rb0 ? angle<1>(sides, dx1, dy1, s1, s2, 'a', 1, -1)
00140 
00141             // not opposite, forming a angle, first b then a,
00142             // directed either both left, or both right
00143             : ra0 && rb1 ? angle<0>(sides, dx1, dy1, s1, s2, 'a', -1, 1)
00144 
00145             // b starts from interior of a
00146             : rb0 ? starts_from_middle(sides, dx1, dy1, s1, s2, 'B', 0, -1)
00147 
00148             // a starts from interior of b (#39)
00149             : ra0 ? starts_from_middle(sides, dx1, dy1, s1, s2, 'A', -1, 0)
00150 
00151             // b ends at interior of a, calculate direction of A from IP
00152             : rb1 ? b_ends_at_middle(sides, dx2, dy2, s1, s2)
00153 
00154             // a ends at interior of b
00155             : ra1 ? a_ends_at_middle(sides, dx1, dy1, s1, s2)
00156 
00157             // normal intersection
00158             : calculate_side<1>(sides, dx1, dy1, s1, s2, 'i', -1, -1)
00159             ;

00281     static inline return_type starts_from_middle(side_info const& sides,
