00148 // There is one thing where centroid is different from e.g. within.
00149 // If the ring has only one point, it might make sense that
00150 // that point is the centroid.
00151 template<typename Point, typename Range>
00152 inline bool range_ok(Range const& range, Point& centroid)
00153 {
00154     std::size_t const n = boost::size(range);
00155     if (n > 1)
00156     {
00157         return true;
00158     }
00159     else if (n <= 0)
00160     {
00161 #if defined(CENTROID_WITH_CATCH)
00162         throw centroid_exception();
00163 #endif
00164         return false;
00165     }
00166     else // if (n == 1)
00167     {
00168         // Take over the first point in a "coordinate neutral way"
00169         copy_coordinates(range.front(), centroid);
00170         return false;
00171     }
00172     return true;
00173 }

00235 template<typename Linestring, typename Point, typename Strategy>
00236 struct centroid_linestring
00237 {
00238     static inline void apply(Linestring const& line, Point& centroid,
00239             Strategy const& strategy)
00240     {
00241         // First version, should
00242         // - be moved to a strategy
00243         // - be made dim-agnostic
00244 
00245         typedef typename point_type<Linestring>::type point_type;
00246         typedef typename boost::range_const_iterator<Linestring>::type point_iterator_type;
00247         typedef segment_iterator<point_iterator_type, point_type> segment_iterator;
00248 
00249         double length = double();
00250         std::pair<double, double> average_sum;
00251 
00252         segment_iterator it(boost::begin(line), boost::end(line));
00253         segment_iterator end(boost::end(line));
00254         while (it != end)
00255         {
00256             double const d = geometry::distance(it->first, it->second);
00257             length += d;
00258 
00259             double const mx = (get<0>(it->first) + get<0>(it->second)) / 2;
00260             double const my = (get<1>(it->first) + get<1>(it->second)) / 2;
00261             average_sum.first += d * mx;
00262             average_sum.second += d * my;
00263             ++it;
00264         }
00265 
00266         set<0>(centroid, average_sum.first / length );
00267         set<1>(centroid, average_sum.second / length );
00268     }
