00077         // Check if iterator is still in box and if not, go to the next section and advance until it is in box
00078         void stay_within_box()
00079         {
00080             // Find section having overlap with specified box
00081             while (m_section_iterator != boost::end(m_sections)
00082                     && ! overlaps(m_section_iterator->bounding_box, m_box))
00083             {
00084                 m_section_iterator++;
00085             }
00086             if (m_section_iterator != boost::end(m_sections))
00087             {
00088                 this->base_reference() = boost::begin(m_ring) + m_section_iterator->begin_index;

00090 
00091                 // While not yet at box, advance
00092                 while(this->base() != m_end
00093                     && detail::preceding<D>(m_section_iterator->directions[0], *this->base(), m_box))

00095                     ++(this->base_reference());

00097

00099                 {
00100                     // This should actually not occur because of bbox check, but to be sure

00103                 }
00104             }
00105             else
00106             {
00107                 this->base_reference() = boost::end(m_ring);
00108                 m_end = boost::end(m_ring);
