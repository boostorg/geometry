00701         // On border -> -2 (left/lower) or 2 (right/upper)
00702         // The only purpose of the value is to not be the same,
00703         // and to denote if it is inside (0)

00709         if (geometry::math::equals(c, left)) return -2;
00710         else if (geometry::math::equals(c, right)) return 2;
00711         else if (c < left) return -1;
00712         else if (c > right) return 1;
00713         else return 0;
