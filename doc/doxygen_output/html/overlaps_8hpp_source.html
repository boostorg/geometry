00041 struct box_box_loop
00042 {
00043     static inline void apply(Box1 const& b1, Box2 const& b2,
00044             bool& overlaps, bool& one_in_two, bool& two_in_one)
00045     {
00046         assert_dimension_equal<Box1, Box2>();
00047 
00048         typedef typename coordinate_type<Box1>::type coordinate_type1;
00049         typedef typename coordinate_type<Box2>::type coordinate_type2;
00050 
00051         coordinate_type1 const& min1 = get<min_corner, Dimension>(b1);
00052         coordinate_type1 const& max1 = get<max_corner, Dimension>(b1);
00053         coordinate_type2 const& min2 = get<min_corner, Dimension>(b2);
00054         coordinate_type2 const& max2 = get<max_corner, Dimension>(b2);
00055 
00056         // We might use the (not yet accepted) Boost.Interval
00057         // submission in the future
00058 
00059         // If:
00060         // B1: |-------|
00061         // B2:           |------|
00062         // in any dimension -> no overlap
00063         if (max1 <= min2 || min1 >= max2)
00064         {
00065             overlaps = false;
00066             return;
00067         }
00068 
00069         // If:
00070         // B1: |--------------------|
00071         // B2:   |-------------|
00072         // in all dimensions -> within, then no overlap
00073         // B1: |--------------------|
00074         // B2: |-------------|
00075         // this is "within-touch" -> then no overlap. So use < and >
00076         if (min1 < min2 || max1 > max2)
00077         {
00078             one_in_two = false;
00079         }
00080         // Same other way round
00081         if (min2 < min1 || max2 > max1)
00082         {
00083             two_in_one = false;
00084         }
00085 
00086         box_box_loop
00087             <
00088                 Box1,
00089                 Box2,
00090                 Dimension + 1,
00091                 DimensionCount
00092             >::apply(b1, b2, overlaps, one_in_two, two_in_one);
00093     }
00094 };
