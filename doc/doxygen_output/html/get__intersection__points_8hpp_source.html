00457             std::size_t Dimension,
00458             typename Box,
00459             typename Sections1, typename Sections2,
00460             typename Map

00463             std::size_t source_id1, Geometry1 const& geometry1,

00465             IntersectionPoints& intersection_points,
00466 
00467             Box const& box,

00470             bool &trivial,

00472     {
00473         // To stop the iteration, fallback to (quadratic) behaviour below certain limits,
00474         // or if dividing does not give any profit.
00475         std::size_t n = sec1.size() + sec2.size();
00476         if (sec1.size() < 5
00477             || sec2.size() < 5

00480         {
00481             intersect(source_id1, geometry1, source_id2, geometry2,
00482                             intersection_points, sec1, sec2, map, trivial);
00483             return;
00484         }

00486         // Divide the complete box in two (alternating) halves
00487         Box lower = box, upper = box;
00488         typename geometry::coordinate_type<Box>::type two = 2.0;
00489         typename geometry::coordinate_type<Box>::type mid
00490             = (geometry::get<min_corner, Dimension>(box)
00491                 + geometry::get<max_corner, Dimension>(box)) / two;

00493         geometry::set<max_corner, Dimension>(lower, mid);
00494         geometry::set<min_corner, Dimension>(upper, mid);

00497         Sections2 lower2, upper2;

00514         divide_and_conquer<1 - Dimension>(source_id1, geometry1,
00515                 source_id2, geometry2, intersection_points,
00516                 lower, lower1, lower2, map, trivial, iteration + 1, n);

00659                 relater::apply(segment, right, seg_id,

00670 private:
00671     template<std::size_t Index, typename Point>
00672     static inline int get_side(Box const& box, Point const& point)
00673     {
00674         // Inside -> 0
00675         // Outside -> -1 (left/below) or 1 (right/above)
00676         // On border -> -2 (left/lower) or 2 (right/upper)
00677         // The only purpose of the value is to not be the same,
00678         // and to denote if it is inside (0)
00679 
00680         typename coordinate_type<Point>::type const& c = get<Index>(point);
00681         typename coordinate_type<Box>::type const& left = get<min_corner, Index>(box);
00682         typename coordinate_type<Box>::type const& right = get<max_corner, Index>(box);
00683 
00684         if (geometry::math::equals(c, left)) return -2;
00685         else if (geometry::math::equals(c, right)) return 2;
00686         else if (c < left) return -1;
00687         else if (c > right) return 1;
00688         else return 0;
00689     }
