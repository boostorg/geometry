00020 // Starting point, specialize basic traits necessary to register a point
00021 // (the 'accessor' is technically not specialization but definition, specialized in another macro)
00022 #define BOOST_GEOMETRY_DETAIL_SPECIALIZE_POINT_TRAITS(Point, Dim, CoordinateType, CoordinateSystem) \
00023     template<> struct tag<Point> { typedef point_tag type; }; \
00024     template<> struct dimension<Point> : boost::mpl::int_<Dim> {}; \
00025     template<> struct coordinate_type<Point> { typedef CoordinateType type; }; \
00026     template<> struct coordinate_system<Point> { typedef CoordinateSystem type; }; \
00027     template<std::size_t Dimension> struct Point##accessor {};

00030 // Non Const version
00031 #define BOOST_GEOMETRY_DETAIL_SPECIALIZE_POINT_ACCESS(Point, CoordinateType) \
00032     template<std::size_t Dimension> struct access<Point, Dimension> { \
00033         static inline CoordinateType get(Point const& p) \
00034         { return Point##accessor<Dimension>::get(p); } \
00035         static inline void set(Point& p, CoordinateType const& value) \
00036         { Point##accessor<Dimension>::set(p, value); } \
00037     };
