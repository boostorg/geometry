00155     static inline void apply(PointOfSegment const& p1,
00156             PointOfSegment const& p2, sums& state)
00157     {
00158         /* Algorithm:
00159         For each segment:
00160         begin
00161             ai = x1 * y2 - x2 * y1;
00162             sum_a2 += ai;
00163             sum_x += ai * (x1 + x2);
00164             sum_y += ai * (y1 + y2);
00165         end
00166         return POINT(sum_x / (3 * sum_a2), sum_y / (3 * sum_a2) )
00167         */
00168 
00169         // Get coordinates and promote them to calculation_type
00170         calculation_type const x1 = boost::numeric_cast<calculation_type>(get<0>(p1));
00171         calculation_type const y1 = boost::numeric_cast<calculation_type>(get<1>(p1));
00172         calculation_type const x2 = boost::numeric_cast<calculation_type>(get<0>(p2));
00173         calculation_type const y2 = boost::numeric_cast<calculation_type>(get<1>(p2));
00174         calculation_type const ai = x1 * y2 - x2 * y1;
00175         state.count++;
00176         state.sum_a2 += ai;
00177         state.sum_x += ai * (x1 + x2);
00178         state.sum_y += ai * (y1 + y2);
00179     }

00181     static inline bool result(sums const& state, Point& centroid)
00182     {
00183         if (state.count > 0 && state.sum_a2 != 0)
00184         {
00185             calculation_type const v3 = 3;
00186             calculation_type const a3 = v3 * state.sum_a2;
00187 
00188             typedef typename geometry::coordinate_type
00189                 <
00190                     Point
00191                 >::type coordinate_type;

00193             set<0>(centroid,
00194                 boost::numeric_cast<coordinate_type>(state.sum_x / a3));
00195             set<1>(centroid,
00196                 boost::numeric_cast<coordinate_type>(state.sum_y / a3));
00197             return true;
