00104 template<typename Range, typename Strategy, std::size_t Minimum>
00105 struct simplify_range
00106 {
00107     static inline void apply(Range const& range, Range& out,
00108                     double max_distance, Strategy const& strategy)
00109     {
00110         // Call do_container for a linestring / ring
00111 
00112         /* For a RING:
00113             The first/last point (the closing point of the ring) should maybe
00114             be excluded because it lies on a line with second/one but last.
00115             Here it is never excluded.
00116 
00117             Note also that, especially if max_distance is too large,
00118             the output ring might be self intersecting while the input ring is
00119             not, although chances are low in normal polygons
00120 
00121             Finally the inputring might have 4 points (=correct),
00122                 the output < 4(=wrong)
00123         */
00124 
00125         if (boost::size(range) <= int(Minimum) || max_distance < 0.0)
00126         {
00127             simplify_copy<Range, Strategy>::apply
00128                 (
00129                     range, out, max_distance, strategy
00130                 );
00131         }
00132         else
00133         {
00134             simplify_range_inserter<Range, Strategy>::apply
00135                 (
00136                     range, std::back_inserter(out), max_distance, strategy
00137                 );
00138         }
00139     }
00140 };

00142 template<typename Polygon, typename Strategy>
00143 struct simplify_polygon
00144 {
00145     static inline void apply(Polygon const& poly_in, Polygon& poly_out,
00146                     double max_distance, Strategy const& strategy)
00147     {
00148         typedef typename ring_type<Polygon>::type ring_type;
00149 
00150         typedef typename boost::range_iterator
00151             <typename interior_type<Polygon>::type>::type iterator_type;
00152         typedef typename boost::range_const_iterator
00153             <typename interior_type<Polygon>::type>::type const_iterator_type;
00154 
00155         // Note that if there are inner rings, and distance is too large,
00156         // they might intersect with the outer ring in the output,
00157         // while it didn't in the input.
00158         simplify_range<ring_type, Strategy, 4>::apply(exterior_ring(poly_in),
00159                         exterior_ring(poly_out),
00160                         max_distance, strategy);
00161 
00162         // Note: here a resizeable container is assumed.
00163         // Maybe we should make this part of the concept.
00164         interior_rings(poly_out).resize(num_interior_rings(poly_in));
00165 
00166         iterator_type it_out = boost::begin(interior_rings(poly_out));
00167 
00168         for (const_iterator_type it_in = boost::begin(interior_rings(poly_in));
00169             it_in != boost::end(interior_rings(poly_in));
00170             ++it_in, ++it_out)
00171         {
00172             simplify_range<ring_type, Strategy, 4>::apply(*it_in,
00173                         *it_out, max_distance, strategy);
00174         }
00175     }
