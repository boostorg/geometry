00441 template
00442 <
00443     typename Geometry1, typename Geometry2,
00444     typename OutputIterator, typename GeometryOut,
00445     int Direction,
00446     typename Strategy
00447 >
00448 struct overlay_and_assemble
00449 {
00450     typedef typename geometry::tag<Geometry1>::type tag1;
00451     typedef typename geometry::tag<Geometry2>::type tag2;
00452     typedef typename geometry::tag<GeometryOut>::type tag_out;
00453 
00454     static inline OutputIterator apply(Geometry1 const& geometry1,
00455                 Geometry2 const& geometry2, OutputIterator out,
00456                 Strategy const& strategy)
00457     {
00458         if (geometry::num_points(geometry1) == 0
00459             || geometry::num_points(geometry2) == 0)
00460         {
00461             return out;
00462         }
00463 
00464         typedef typename geometry::point_type<GeometryOut>::type point_type;
00465 
00466 
00467         typedef detail::overlay::traversal_turn_info<point_type> turn_info;
00468         typedef std::deque<turn_info> container_type;
00469 
00470         // "Abuse" rangetype for ringtype: for polygon, it is the type of the
00471         // exterior ring. For ring, it is the ring itself. That is what is
00472         // wished here as well.
00473         typedef typename geometry::range_type<GeometryOut>::type ring_type;
00474         typedef std::vector<ring_type> hole_vector_type;
00475 
00476         container_type ips; // intersection points
00477 
00478         boost::geometry::get_turns<detail::overlay::CalculateDistancePolicy>(geometry1, geometry2, ips);
00479         bool had_ips = ips.size() > 0;
00480 
00481         // Gather non-intersecting exterior / interior rings
00482         // Map from [source_index,ring_index] to code
00483         //      [1=intersecting, 2=intersecting after merging]
00484         typedef std::map<std::pair<int, int>, int> intersect_map;
00485         intersect_map intersect_codes;
00486 
00487         // Get codes (before merging)
00488         get_intersect_codes<tag1>::apply(geometry1, 0, ips, intersect_codes, 1);
00489         get_intersect_codes<tag2>::apply(geometry2, 1, ips, intersect_codes, 1);
00490 
00491         typename Strategy::side_strategy_type side_strategy;
00492 
00493         // Get new codes (after merging)
00494         get_intersect_codes<tag1>::apply(geometry1, 0, ips, intersect_codes, 2);
00495         get_intersect_codes<tag2>::apply(geometry2, 1, ips, intersect_codes, 2);
00496 
00497         // Doc to be updated and moved to Paper!
00498         // If there are no IP-s at all, check if one point is in other geometry
00499         // (both geometries have points, see check above)
00500         // If yes, take the correct one
00501 
00502         // (which depends on intersection/union)
00503         // If both are unrelated, take none (intersection) or both (union)
00504 
00505         // If there are disjoint or containing holes:
00506         // - Hole of A can be within hole of B
00507         //   - for union: take hole of A
00508         //   - for intersection: take hole of B
00509         // - Hole of A can be disjoint from B
00510         //   - for union: remove those holes
00511         //   - for intersection: take all those holes
00512         // -> generic procedure
00513         // - Collect all holes in one vector
00514         // - Sort them on decreasing area
00515         // - Check if one is within a larger one,
00516         //   - if yes, take one of them
00517         //   - if no, take all, or none of them
00518 
00519         // END DOC
00520 
00521         // Gather disjoint or equal holes
00522         hole_vector_type ni_holes;
00523         get_disjoint_or_equal_holes<tag1>::apply(Direction, geometry1, 0, intersect_codes, ni_holes);
00524         get_disjoint_or_equal_holes<tag2>::apply(Direction, geometry2, 1, intersect_codes, ni_holes);
00525 
00526         if (! had_ips && ips.size() <= 0)
00527         {
00528             // No intersections or collinearies found
00529             GeometryOut result;
00530             if (take_one<Direction>(geometry1, geometry2, result))
00531             {
00532                 add_holes
00533                     <
00534                         tag_out, Geometry1, GeometryOut, hole_vector_type,
00535                         Direction
00536                     >::apply(geometry1, result, ni_holes);
00537                 *out = result;
00538                 out++;
00539             }
00540             else if (take_one<Direction>(geometry2, geometry1, result))
00541             {
00542                 add_holes
00543                     <
00544                         tag_out,
00545                         Geometry2, GeometryOut, hole_vector_type,
00546                         Direction
00547                     >::apply(geometry2, result, ni_holes);
00548                 *out = result;
00549                 out++;
00550             }
00551             else
00552             {
00553                 take_none_or_both
00554                     <
00555                         Geometry1, Geometry2,
00556                         OutputIterator, GeometryOut,
00557                         Direction
00558                     >::apply(geometry1, geometry2, out);
00559             }
00560             return out;
00561         }
00562 
00563 
00564 //std::cout << "enrich" << std::endl;
00565         geometry::enrich_intersection_points(ips, geometry1, geometry2, side_strategy);
00566 
00567         std::vector<ring_type> rings;
00568 //std::cout << "traverse" << std::endl;
00569         geometry::traverse<typename Strategy::side_strategy_type, ring_type>
00570             (
00571                 geometry1,
00572                 geometry2,
00573                 Direction == -1 
00574                     ? boost::geometry::detail::overlay::operation_intersection
00575                     : boost::geometry::detail::overlay::operation_union
00576                     ,
00577                 ips,
00578                 std::back_inserter(rings)
00579             );
00580 //std::cout << "assemble" << std::endl;
00581 
00582         // Get exterior ring (only done if exterior rings are equal or inside each other)
00583         //if (! trivial)
00584         {
00585             get_exterior_ring<tag1, tag2>::apply(geometry1, geometry2, intersect_codes, rings, Direction);
00586         }
00587 
00588         switch (boost::size(rings))
00589         {
00590             case 0 : break;
00591             case 1 :
00592                 {
00593                     GeometryOut result;
00594                     geometry::convert(rings.front(), result);
00595                     add_holes
00596                         <
00597                             tag_out,
00598                             ring_type, GeometryOut, hole_vector_type,
00599                             Direction
00600                         >::apply(rings.front(), result, ni_holes);

00605             default :

00607                     <
00608                         tag_out, GeometryOut,
