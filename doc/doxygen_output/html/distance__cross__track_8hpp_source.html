00075                 PointOfSegment const& sp1, PointOfSegment const& sp2) const
00076     {
00077         // http://williams.best.vwh.net/avform.htm#XTE
00078         double d1 = m_strategy.apply(sp1, p);
00079 
00080         // Actually, calculation of d2 not necessary if we know that the projected point is on the great circle...
00081         double d2 = m_strategy.apply(sp2, p);
00082 
00083         double crs_AD = course(sp1, p);
00084         double crs_AB = course(sp1, sp2);
00085         double XTD = std::abs(asin(sin(d1) * sin(crs_AD - crs_AB)));
00086

00090 
00091 
00092         // Return shortest distance, either to projected point on segment sp1-sp2, or to sp1, or to sp2
00093         return return_type(m_radius * (std::min)((std::min)(d1, d2), XTD));
00094     }

00103     inline double course(Point const& p1, Point const& p2) const
00104     {
00105         // http://williams.best.vwh.net/avform.htm#Crs
00106         double dlon = get_as_radian<0>(p2) - get_as_radian<0>(p1);
00107         double cos_p2lat = cos(get_as_radian<1>(p2));
00108 
00109         // "An alternative formula, not requiring the pre-computation of d"
00110         return atan2(sin(dlon) * cos_p2lat,
00111             cos(get_as_radian<1>(p1)) * sin(get_as_radian<1>(p2))
00112             - sin(get_as_radian<1>(p1)) * cos_p2lat * cos(dlon));
00113     }
