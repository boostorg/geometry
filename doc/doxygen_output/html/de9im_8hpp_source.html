00032     static inline return_type rays_intersect(bool on_segment,
00033                     double ra, double rb,
00034                     coordinate_type const& dx1, coordinate_type const& dy1,
00035                     coordinate_type const& dx2, coordinate_type const& dy2,
00036                     coordinate_type const& wx, coordinate_type const& wy,
00037                     S1 const& s1, S2 const& s2)
00038     {
00039         if(on_segment)
00040         {
00041             // 0 <= ra <= 1 and 0 <= rb <= 1
00042             // Now check if one of them is 0 or 1, these are "touch" cases
00043             bool a = math::equals(ra, 0.0) || math::equals(ra, 1.0);
00044             bool b = math::equals(rb, 0.0) || math::equals(rb, 1.0);
00045             if (a && b)
00046             {
00047                 // Touch boundary/boundary: i-i == -1, i-b == -1, b-b == 0
00048                 // Opposite: if both are equal they touch in opposite direction
00049                 return de9im_segment(ra,rb,
00050                         -1, -1, 1,
00051                         -1,  0, 0,
00052                          1,  0, 2, false, math::equals(ra,rb));
00053             }
00054             else if (a || b)
00055             {
00056                 // Touch boundary/interior: i-i == -1, i-b == -1 or 0, b-b == -1
00057                 int A = a ? 0 : -1;
00058                 int B = b ? 0 : -1;
00059                 return de9im_segment(ra,rb,
00060                         -1,  B, 1,
00061                          A, -1, 0,
00062                          1,  0, 2);
00063             }
00064 
00065             // Intersects: i-i == 0, i-b == -1, i-e == 1
00066             return de9im_segment(ra,rb,
00067                      0, -1, 1,
00068                     -1, -1, 0,
00069                      1,  0, 2);
00070         }
00071 
00072         // Not on segment, disjoint
00073         return de9im_segment(ra,rb,
00074                 -1, -1, 1,
00075                 -1, -1, 0,
00076                  1,  0, 2);
00077     }
