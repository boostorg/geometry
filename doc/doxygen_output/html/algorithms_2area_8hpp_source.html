00065 template<typename Box, typename Strategy>
00066 struct box_area
00067 {
00068     typedef typename coordinate_type<Box>::type return_type;
00069 
00070     static inline return_type apply(Box const& box, Strategy const&)
00071     {
00072         // Currently only works for 2D Cartesian boxes
00073         assert_dimension<Box, 2>();
00074 
00075         return_type const dx = get<max_corner, 0>(box)
00076                 - get<min_corner, 0>(box);
00077         return_type const dy = get<max_corner, 1>(box)
00078                 - get<min_corner, 1>(box);
00079 
00080         return dx * dy;
00081     }

00101 template<typename Ring, typename Strategy>
00102 struct ring_area<Ring, clockwise, Strategy>
00103 {
00104     BOOST_CONCEPT_ASSERT( (geometry::concept::AreaStrategy<Strategy>) );
00105 
00106     typedef typename Strategy::return_type type;
00107 
00108     static inline type apply(Ring const& ring, Strategy const& strategy)
00109     {
00110         assert_dimension<Ring, 2>();
00111 
00112         // A closed linear_ring has at least four points,
00113         // if not, there is no (zero) area
00114         if (boost::size(ring) < 4)
00115         {
00116             return type();
00117         }
00118 
00119         typedef typename boost::range_const_iterator<Ring>::type iterator_type;
00120 
00121         typename Strategy::state_type state;
00122 
00123         iterator_type it = boost::begin(ring);
00124         for (iterator_type previous = it++;
00125             it != boost::end(ring);
00126             previous = it++)
00127         {
00128             strategy.apply(*previous, *it, state);
00129         }
00130         return strategy.result(state);
00131 
00132     }
