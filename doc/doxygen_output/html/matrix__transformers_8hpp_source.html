00078     inline bool apply(P1 const& p1, P2& p2) const
00079     {
00080         assert_dimension_greater_equal<P1, 2>();
00081         assert_dimension_greater_equal<P2, 2>();
00082 
00083         coordinate_type const& c1 = get<0>(p1);
00084         coordinate_type const& c2 = get<1>(p1);
00085 
00086         typedef typename geometry::coordinate_type<P2>::type ct2;
00087 
00088         set<0>(p2, boost::numeric_cast<ct2>(
00089             c1 * m_matrix(0,0) + c2 * m_matrix(0,1) + m_matrix(0,2)));
00090         set<1>(p2, boost::numeric_cast<ct2>(
