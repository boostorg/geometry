00089 template <typename Ring1, typename Ring2>
00090 struct ring_ring
00091 {
00092     static inline bool apply(Ring1 const& ring1, Ring2 const& ring2, bool check_area = true)
00093     {
00094         // Note: this implementation makes use of getting interections
00095         // and merge them. If all IP's disappear, the ring should be the same
00096         // (because merging removes collinear or non-collinear
00097         // IP's following the same path)
00098         // However, this implementation should be redone using
00099         // a linear time algorithm (getting left-most points of both
00100         // and follow them using circular iterator and distance/side)
00101 
00102         // obvious check, area's should be the same.
00103         if (check_area && geometry::area(ring1) != geometry::area(ring2))
00104         {
00105             return false;
00106         }
00107         // We could check more (perimeter,centroid,envelope)
00108         // For now we go directly to intersection points
00109 
00110         typedef typename geometry::point_type<Ring1>::type point_type;
00111         typedef detail::intersection::intersection_point<point_type> ip_type;
00112         typedef std::deque<ip_type> container_type;
00113 
00114         container_type ips;
00115 
00116 
00117         geometry::get_intersection_points(ring1, ring2, ips);
00118         if (ips.size() == 0)
00119         {
00120             return false;
00121         }
00122         geometry::merge_intersection_points(ips);
00123         if (ips.size() == 0)
00124         {
00125             // All IP's are merged. Meaning collinear / path can be followed
00126             return true;
00127         }
00128 
00129         return false;
00130     }
00131 };

00181     static inline bool apply(Polygon1 const& polygon1, Polygon2 const& polygon2)
00182     {
00183         // Check number of rings (area check is done in exterior ring check)
00184         if (geometry::num_interior_rings(polygon1) != geometry::num_interior_rings(polygon2))
00185         {
00186             return false;
00187         }
00188 
00189         typedef typename geometry::ring_type<Polygon1>::type ring_type1;
00190         typedef typename geometry::ring_type<Polygon2>::type ring_type2;
00191         typedef ring_ring<ring_type1, ring_type2> compare;
00192 
00193         // Check exterior ring
00194         if (! compare::apply(exterior_ring(polygon1), exterior_ring(polygon2)))
00195         {
00196             return false;
00197         }

00200         // for performance reasons (area is not re-calculated in ring-compare)

00211                 = interior_rings(polygon2);
00212 
00213         // Compare all rings (having equal area)
00214         for (typename std::vector<sortable>::iterator it1 = int1.begin();
00215             it1 != int1.end();
00216             ++it1)
00217         {
00218             for (typename std::vector<sortable>::iterator it2 = int2.begin();

00220                     && it2 != int2.end()

00225                     && geometry::math::equals(it1->area, it2->area))

00234                 }
