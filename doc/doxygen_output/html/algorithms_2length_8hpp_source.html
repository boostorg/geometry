00074 template<typename Range, typename Strategy>
00075 struct range_length
00076 {
00077     typedef typename length_result<Range>::type return_type;
00078 
00079     static inline return_type apply(
00080             Range const& range, Strategy const& strategy)
00081     {
00082         return_type sum = return_type();
00083 
00084         typedef typename boost::range_const_iterator<Range>::type iterator_type;
00085         iterator_type it = boost::begin(range);
00086         if (it != boost::end(range))
00087         {
00088             iterator_type previous = it++;
00089             while(it != boost::end(range))
00090             {
00091                 // Add point-point distance using the return type belonging
00092                 // to strategy
00093                 sum += strategy.apply(*previous, *it);
00094                 previous = it++;
00095             }
00096         }
00097 
00098         return sum;
00099     }
