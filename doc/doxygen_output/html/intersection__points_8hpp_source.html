00063         promoted_type const wx = get<0, 0>(s1) - get<0, 0>(s2);

00065         promoted_type const d = (dy2 * dx1) - (dx2 * dy1);

00068         // r: ratio 0-1 where intersection divides A/B
00069         promoted_type const r = da / d;

00071         result.count = 1;
00072         set<0>(result.intersections[0],
00073             boost::numeric_cast<coordinate_type>(s1x + r * promoted_type(dx1)));
00074         set<1>(result.intersections[0],
00075             boost::numeric_cast<coordinate_type>(s1y + r * promoted_type(dy1)));
