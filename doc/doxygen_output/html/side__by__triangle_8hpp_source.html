00028         // Check at which side of a segment a point lies:
00029         // left of segment (> 0), right of segment (< 0), on segment (0)
00030         // In fact this is twice the area of a triangle
00031         template <typename CalculationType>
00032         struct side_by_triangle
00033         {
00034 
00035             // Template member function, because it is not always trivial
00036             // or convenient to explicitly mention the typenames in the
00037             // strategy-struct itself.
00038 
00039             // Types can be all three different. Therefore it is
00040             // not implemented (anymore) as "segment"
00041 
00042             template <typename P1, typename P2, typename P>
00043             static inline int apply(P1 const& p1, P2 const& p2, P const& p)
00044             {
00045                 typedef typename boost::mpl::if_c
00046                     <
00047                         boost::is_void<CalculationType>::type::value,
00048                         typename select_most_precise
00049                             <
00050                                 typename select_most_precise
00051                                     <
00052                                         typename coordinate_type<P1>::type,
00053                                         typename coordinate_type<P2>::type
00054                                     >::type,
00055                                 typename coordinate_type<P>::type
00056                             >::type,
00057                         CalculationType
00058                     >::type coordinate_type;
00059 
00060 //std::cout << "side: " << typeid(coordinate_type).name() << std::endl;
00061                 coordinate_type const x = get<0>(p);
00062                 coordinate_type const y = get<1>(p);
00063 
00064                 coordinate_type const sx1 = get<0>(p1);
00065                 coordinate_type const sy1 = get<1>(p1);
00066                 coordinate_type const sx2 = get<0>(p2);
00067                 coordinate_type const sy2 = get<1>(p2);
00068 
00069                 // Promote float->double, small int->int
00070                 typedef typename select_most_precise
00071                     <
00072                         coordinate_type, 
00073                         double
00074                     >::type promoted_type;
00075 
00076                 promoted_type const dx = sx2 - sx1;
00077                 promoted_type const dy = sy2 - sy1;
00078                 promoted_type const dpx = x - sx1;
00079                 promoted_type const dpy = y - sy1;
00080 
00081                 promoted_type const s = dx * dpy - dy * dpx;
00082 
00083                 return math::equals(s, 0) ? 0 : s > 0 ? 1 : -1;
00084                 //return s > 0 ? 1 : s < 0 ? -1 : 0;
00085             }
00086         };
