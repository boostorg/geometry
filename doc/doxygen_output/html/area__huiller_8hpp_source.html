00065             return - sum * radius * radius;
00066         }

00078     inline void apply(PointOfSegment const& p1,
00079                 PointOfSegment const& p2,
00080                 excess_sum& state) const
00081     {
00082         if (! geometry::math::equals(get<0>(p1), get<0>(p2)))
00083         {
00084             // Distance p1 p2
00085             double a = state.distance_over_unit_sphere.apply(p1, p2);
00086 
00087             // Sides on unit sphere to south pole
00088             double b = 0.5 * math::pi - geometry::get_as_radian<1>(p2);
00089             double c = 0.5 * math::pi - geometry::get_as_radian<1>(p1);
00090 
00091             // Semi parameter
00092             double s = 0.5 * (a + b + c);
00093 
00094             // E: spherical excess, using l'Huiller's formula
00095             // [tg(e / 4)]2   =   tg[s / 2]  tg[(s-a) / 2]  tg[(s-b) / 2]  tg[(s-c) / 2]
00096             double E = 4.0 * atan(sqrt(std::abs(tan(s / 2)
00097                     * tan((s - a) / 2)
00098                     * tan((s - b) / 2)
00099                     * tan((s - c) / 2))));
00100 
00101             E = std::abs(E);
00102 
00103             // In right direction: positive, add area. In left direction: negative, subtract area.
00104             // Longitude comparisons are not so obvious. If one is negative, other is positive,
00105             // we have to take the dateline into account.
00106             // TODO: check this / enhance this, should be more robust. See also the "grow" for ll
00107             // TODO: use minmax or "smaller"/"compare" strategy for this
00108             double lon1 = geometry::get_as_radian<0>(p1) < 0
00109                 ? geometry::get_as_radian<0>(p1) + math::two_pi
00110                 : geometry::get_as_radian<0>(p1);
00111 
00112             double lon2 = geometry::get_as_radian<0>(p2) < 0
00113                 ? geometry::get_as_radian<0>(p2) + math::two_pi
00114                 : geometry::get_as_radian<0>(p2);
00115 
00116             if (lon2 < lon1)
00117             {
00118                 E = -E;
00119             }
00120 
00121             state.sum += E;
00122         }
00123     }
00124 
00125     inline return_type result(excess_sum const& state) const
00126     {
00127         return state.area(m_radius);
00128     }
