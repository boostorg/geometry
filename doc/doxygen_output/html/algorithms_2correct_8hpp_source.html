00037 template <typename Box, std::size_t Dimension, std::size_t DimensionCount>
00038 struct correct_box_loop
00039 {
00040     typedef typename coordinate_type<Box>::type coordinate_type;
00041 
00042     static inline void apply(Box& box)
00043     {
00044         if (get<min_corner, Dimension>(box) > get<max_corner, Dimension>(box))
00045         {
00046             // Swap the coordinates
00047             coordinate_type max_value = get<min_corner, Dimension>(box);
00048             coordinate_type min_value = get<max_corner, Dimension>(box);
00049             set<min_corner, Dimension>(box, min_value);
00050             set<max_corner, Dimension>(box, max_value);
00051         }

00088 // close a linear_ring, if not closed
00089 template <typename Ring, typename Predicate>
00090 struct correct_ring
00091 {
00092     typedef typename point_type<Ring>::type point_type;
00093 
00094     typedef typename strategy_area
00095         <
00096             typename cs_tag<point_type>::type,
00097             point_type
00098         >::type strategy_type;
00099 
00100     typedef detail::area::ring_area
00101             <
00102                 Ring,
00103                 geometry::point_order<Ring>::value,
00104                 strategy_type
00105             > ring_area_type;
00106 
00107 
00108     static inline void apply(Ring& r)
00109     {
00110         // Check close-ness
00111         if (boost::size(r) > 2)
00112         {
00113             // check if closed, if not, close it
00114             if (geometry::disjoint(*boost::begin(r), *(boost::end(r) - 1)))
00115             {
00116                 point_type first;
00117                 geometry::copy_coordinates(*boost::begin(r), first);
00118                 *(std::back_inserter(r)++) = first;
00119             }
00120         }
00121         // Check area
00122         Predicate predicate;
00123         if (predicate(ring_area_type::apply(r, strategy_type()), 0))
00124         {
00125             std::reverse(boost::begin(r), boost::end(r));
00126         }
00127     }
00128 };

00132 template <typename Polygon>

00138     {
00139         correct_ring<ring_type, std::less<double> >::apply(exterior_ring(poly));
