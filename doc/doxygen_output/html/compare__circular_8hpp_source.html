00050 template <typename CoordinateType, typename Units, typename Compare>
00051 struct circular_comparator
00052 {
00053     static inline CoordinateType put_in_range(CoordinateType const& c,
00054             double min_border, double max_border)
00055     {
00056         CoordinateType value = c;
00057         while (value < min_border)
00058         {
00059             value += detail::shift<Units>::full();
00060         }
00061         while (value > max_border)
00062         {
00063             value -= detail::shift<Units>::full();
00064         }
00065         return value;
00066     }
00067 
00068     inline bool operator()(CoordinateType const& c1, CoordinateType const& c2)  const
00069     {
00070         Compare compare;
00071 
00072         // Check situation that one of them is e.g. std::numeric_limits.
00073         static const double full = detail::shift<Units>::full();
00074         double mx = 10.0 * full;
00075         if (c1 < -mx || c1 > mx || c2 < -mx || c2 > mx)
00076         {
00077             // do normal comparison, using circular is not useful
00078             return compare(c1, c2);
00079         }
00080 
00081         static const double half = full / 2.0;
00082         CoordinateType v1 = put_in_range(c1, -half, half);
00083         CoordinateType v2 = put_in_range(c2, -half, half);
00084 
00085         // Two coordinates on a circle are
00086         // at max <= half a circle away from each other.
00087         // So if it is more, shift origin.
00088         CoordinateType diff = std::abs(v1 - v2);
00089         if (diff > half)
00090         {
00091             v1 = put_in_range(v1, 0, full);
00092             v2 = put_in_range(v2, 0, full);
00093         }
00094 
00095         return compare(v1, v2);
00096     }
00097 };
