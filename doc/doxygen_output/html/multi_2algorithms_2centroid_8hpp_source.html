00063     static inline void apply(Multi const& multi, Point& centroid,
00064             Strategy const& strategy)
00065     {
00066         // If there is nothing in any of the ranges, it is not possible
00067         // to calculate the centroid
00068         if (geometry::num_points(multi) == 0)
00069         {
00070             throw centroid_exception();

00072 
00073         typename Strategy::state_type state;
00074 
00075         for (typename boost::range_const_iterator<Multi>::type
00076                 it = boost::begin(multi);
00077             it != boost::end(multi);
00078             ++it)
00079         {
00080             Policy::apply(*it, strategy, state);
00081         }
