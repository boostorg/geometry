00122     typename Turn,
00123     typename Iterator
00124 >
00125 inline bool select_next_ip(operation_type operation,
00126             Turn& turn,
00127             segment_identifier const& seg_id,
00128             Iterator& selected)
00129 {
00130     bool has_tp = false;
00131     selected = boost::end(turn.turns);
00132     for (Iterator it = boost::begin(turn.turns); 
00133         it != boost::end(turn.turns); 
00134         ++it)
00135     {
00136         // In some cases there are two alternatives.
00137         // For "ii", take the other one (alternate)
00138         // For "uu", take the same one
00139         // For "cc", take either one, but if there is a starting one,
00140         //           take that one.
00141         if (   (it->operation == operation_continue
00142                 && (! has_tp 
00143                     || it->visited.started()
00144                     )
00145                 )
00146             || (it->operation == operation
00147                 && (! has_tp 
00148                     || select_source(operation, 
00149                             it->seg_id.source_index, seg_id.source_index)
00150                     )
00151                 )
00152             )
00153         {
00154             selected = it;
00155             has_tp = true;
00156         }
00157     }
00158 
00159     return has_tp;
00160 }
