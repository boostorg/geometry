00149     template <typename P>
00150     inline void spherical_to_cartesian(double phi, double theta, double r, P& p)
00151     {
00152         assert_dimension<P, 3>();
00153 
00154         // http://en.wikipedia.org/wiki/List_of_canonical_coordinate_transformations#From_spherical_coordinates
00155         // Phi = first, theta is second, r is third, see documentation on cs::spherical
00156         double const sin_theta = std::sin(theta);
00157         set<0>(p, r * sin_theta * std::cos(phi));

00159         set<2>(p, r * std::cos(theta));

00188     {
00189         assert_dimension<P, 3>();
00190 
00191         // http://en.wikipedia.org/wiki/List_of_canonical_coordinate_transformations#From_Cartesian_coordinates
00192         double const r = std::sqrt(x * x + y * y + z * z);
00193         set<2>(p, r);
