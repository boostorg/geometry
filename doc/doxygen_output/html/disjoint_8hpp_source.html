00062 struct general
00063 {
00064     static inline bool apply(Geometry1 const& geometry1, Geometry2 const& geometry2)
00065     {
00066 
00067         typedef typename geometry::point_type<Geometry1>::type point_type;
00068         typedef detail::intersection::intersection_point<point_type> ip_type;
00069         std::deque<ip_type> ips; // intersection points
00070 
00071         // Get any intersection
00072         geometry::get_intersection_points(geometry1, geometry2, ips);
00073         if (ips.size() > 0)
00074         {
00075             return false;
00076         }
00077         // If there is no intersection of segments, they might ly
00078         // inside each other
00079         point_type p1;
00080         geometry::point_on_border(geometry1, p1);
00081         if (geometry::within(p1, geometry2))
00082         {
00083             return false;
00084         }
00085 
00086         typename geometry::point_type<Geometry1>::type p2;
00087         geometry::point_on_border(geometry2, p2);
00088         if (geometry::within(p2, geometry1))
00089         {
00090             return false;
00091         }
00092 
00093         return true;
00094     }
