00185         // Clip the segment. Five situations:
00186         // 1. Segment is invisible, finish line if any (shouldn't occur)
00187         // 2. Segment is completely visible. Add (p1)-p2 to line
00188         // 3. Point 1 is invisible (clipped), point 2 is visible. Start new line from p1-p2...
00189         // 4. Point 1 is visible, point 2 is invisible (clipped). End the line with ...p2
00190         // 5. Point 1 and point 2 are both invisible (clipped). Start/finish an independant line p1-p2
00191         //
00192         // This results in:
00193         // a. if p1 is clipped, start new line
00194         // b. if segment is partly or completely visible, add the segment
00195         // c. if p2 is clipped, end the line
00196 
00197         bool c1 = false;
00198         bool c2 = false;
00199         segment<point_type> s(p1, p2);
00200 
00201         if (!strategy.clip_segment(b, s, c1, c2))
00202         {
00203             strategy.apply(line_out, out);
00204         }
00205         else
00206         {
00207             // a. If necessary, finish the line and add a start a new one
00208             if (c1)
00209             {
00210                 strategy.apply(line_out, out);
00211             }
00212 
00213             // b. Add p1 only if it is the first point, then add p2
00214             if (boost::empty(line_out))
